\section*{Exercise 1}
\subsection*{(a)}
\begin{align*}
	\mathbb{E}[\|Z - \mathbb{E}[Z]\|_2^2] 
	&= \mathbb{E}[\|Z\|_2^2 - 2 \langle Z, \mathbb{E}[Z] \rangle + \|\mathbb{E}[Z]\|_2^2] \\
	&= \mathbb{E}[\|Z\|_2^2] - 2 \mathbb{E}[Z]^T \mathbb{E}[Z] + \|\mathbb{E}[Z]\|_2^2 \\
	&= \mathbb{E}[\|Z\|]_2^2 - \|\mathbb{E}[Z]\|_2^2.
\end{align*}
